FormControllerViewModel

abstract class FormControllerViewModel<Model>(var emptyModel: () -> Model) : ViewModel, FormController<Model>

View Model for controlling forms

In an app which uses ViewModel, each screen ViewModel can inherit from this class to separate business logic from the UI layer.

Parameters

Model

The type which the form populates

import com.idscodelabs.compose_form.examples.helpers.ExampleScreen

fun main() { 
   //sampleStart 
   ExampleScreen {
        val viewModel =
            viewModel { FormViewModelExampleViewModel() }
        val uiState by viewModel.uiState.collectAsState()

        Box(Modifier.fillMaxSize()) {
            if (uiState.loading) {
                CircularProgressIndicator(Modifier.align(Alignment.Center))
            }
            if (uiState.text.isNullOrEmpty()) {
                Form(controller = viewModel) {
                    FormTextField(
                        enabled = !uiState.loading,
                        modelProperty = FormTextFieldExampleModel::value,
                        validator = NotEmptyValidator(),
                        updateModel = { value = it },
                    ) {
                        DefaultTextEntry(hint = "Value")
                    }
                    Button(
                        enabled = !uiState.loading,
                        modifier = Modifier.fillMaxWidth(),
                        onClick = viewModel::onSubmitPressed,
                    ) {
                        Text("Submit")
                    }
                }
            } else {
                Text("Result is ${uiState.text}")
            }
        }
    } 
   //sampleEnd
}
